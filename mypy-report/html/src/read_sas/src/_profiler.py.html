from __future__ import annotations
import cProfile
import functools
import threading
from types import TracebackType
from pathlib import Path
from typing import Callable, TypeVar

T = TypeVar("T")


class Profiler:
    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super(Profiler, cls).__new__(cls)
                    cls._instance.profiler = cProfile.Profile()
        return cls._instance

    def __enter__(self) -> "Profiler":
        """Enable the profiler inside a context manager."""
        self._instance.enable()
        return self

    def __exit__(
        self,
        exc_type: type[BaseException] | None,
        exc_val: BaseException | None,
        exc_tb: TracebackType | None,
    ) -> None:
        """Disable the profiler and print the stats."""
        self.profiler.disable()
        self.profiler.print_stats(sort="tottime")

    def print_stats(
        self, sort: str = "tottime", file: str | Path | None = None
    ) -> None:
        """Print the profiler stats."""
        self.profiler.print_stats(sort=sort, stream=file)

    def reset(self) -> None:
        """Clear the profiler stats."""
        self.profiler.clear_stats()

    def profile(
        self, sort: str = "tottime", file: str | Path | None = None
    ) -> Callable[[Callable[..., T]], Callable[..., T]]:
        """Decorate a function to profile that function."""

        def decorator(func: Callable[..., T]) -> Callable[..., T]:
            @functools.wraps(func)
            def wrapper(*args, **kwargs) -> T:
                self.profiler.enable()
                result = func(*args, **kwargs)
                self.profiler.disable()
                self.print_stats(sort=sort, file=file)
                return result

            return wrapper

        return decorator
